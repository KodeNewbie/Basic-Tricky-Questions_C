2.write a logic to swap two numbers with using third variable

for example a= 5, b = 6 
output as a = 6 , b = 5
----------------
#include <stdio.h>
void main()
{
    int a = 5, b = 6, c;

    printf("Given value of a,b : %d,%d", a, b);

    c = b;

    b = a;
    a = c;

    printf("New value of a,b : %d,%d", a, b);
}
